## 3. 依赖注入（DI）

### 服务生命周期

| 生命周期 | 说明 | 适用场景 |
|---------|------|---------|
| `Transient` | 每次请求创建新实例 | 无状态服务 |
| `Scoped` | 每个作用域一个实例 | Web 请求、数据库上下文 |
| `Singleton` | 整个应用只有一个实例 | 缓存、配置服务 |

### 注册服务

```
csharp
// Program.cs 或 Startup.cs
var builder = WebApplication.CreateBuilder(args);

// ① Transient - 每次都新建
builder.Services.AddTransient<IEmailService, EmailService>();

// ② Scoped - 每个请求一个实例
builder.Services.AddScoped<IDatabaseContext, DatabaseContext>();

// ③ Singleton - 单例
builder.Services.AddSingleton<ICacheService, CacheService>();

// ④ 工厂模式注册
builder.Services.AddSingleton<IMessageService>(sp =>
{
    var config = sp.GetRequiredService<IConfiguration>();
    return new MessageService(config["ApiKey"]);
});

// ⑤ 泛型注册
builder.Services.AddScoped(typeof(IRepository<>), typeof(Repository<>));
```
### 使用服务

```
csharp

## 6. 异步编程

### async/await 基础

```
csharp
// 异步方法签名
public async Task<string> GetDataAsync()
{
    // 模拟异步操作
    await Task.Delay(1000);
    return "Data";
}

// 并行执行多个任务
public async Task ProcessMultipleAsync()
{
    var task1 = GetDataAsync();
    var task2 = GetOtherDataAsync();
    
    // 等待所有任务完成
    await Task.WhenAll(task1, task2);
    
    var result1 = await task1;
    var result2 = await task2;
}

// 超时控制
public async Task WithTimeoutAsync()
{
    using var cts = new CancellationTokenSource(TimeSpan.FromSeconds(5));
    
    try
    {
        var result = await GetDataAsync(cts.Token);
    }
    catch (OperationCanceledException)
    {
        Console.WriteLine("操作超时");
    }
}
```
### Task 最佳实践

```
csharp
// ✅ 正确：异步方法返回 Task
public async Task<List<TaskItem>> GetTasksAsync()
{
    return await _dbContext.Tasks.ToListAsync();
}

// ✅ 正确：带取消令牌
public async Task<List<TaskItem>> GetTasksAsync(CancellationToken ct = default)
{
    return await _dbContext.Tasks.ToListAsync(ct);
}

// ❌ 错误：避免 async void（除事件处理外）
public async void BadMethod() { }

// ❌ 错误：避免 Result/Wait（会阻塞线程）
var result = GetDataAsync().Result; // 可能导致死锁
GetDataAsync().Wait(); // 阻塞线程
```
### WPF 中的异步

```
csharp
public class MainViewModel : INotifyPropertyChanged
{
    private readonly ITaskService _taskService;
    
    public ICommand LoadTasksCommand { get; }
    
    public MainViewModel(ITaskService taskService)
    {
        _taskService = taskService;
        LoadTasksCommand = new RelayCommand(async _ => await LoadTasksAsync());
    }
    
    private async Task LoadTasksAsync()
    {
        IsLoading = true;
        
        try
        {
            // UI 线程切换到后台线程执行
            Tasks = await _taskService.GetAllAsync();
        }
        catch (Exception ex)
        {
            ErrorMessage = ex.Message;
        }
        finally
        {
            IsLoading = false;
        }
    }
    
    private bool isLoading;
    public bool IsLoading
    {
        get => isLoading;
        set { isLoading = value; OnPropertyChanged(); }
    }
}

## 7. LINQ 与集合操作

### 查询语法 vs 方法语法

```
csharp
var tasks = new List<TaskItem>
{
    new() { Title = "Task 1", Priority = 1, IsDone = false },
    new() { Title = "Task 2", Priority = 2, IsDone = true },
    new() { Title = "Task 3", Priority = 3, IsDone = false }
};

// 查询语法
var query = from t in tasks
            where !t.IsDone
            orderby t.Priority descending
            select t;

// 方法语法（推荐）
var method = tasks
    .Where(t => !t.IsDone)
    .OrderByDescending(t => t.Priority)
    .ToList();
```
### 高级 LINQ 操作

```
csharp
// 分组 + 投影
var grouped = tasks
    .GroupBy(t => t.Category)
    .Select(g => new
    {
        Category = g.Key,
        Count = g.Count(),
        AvgPriority = g.Average(t => t.Priority),

    })
    .ToList();

// Join 连接
var users = new List<User> { ... };
var userTasks = tasks
    .Join(users,
        t => t.UserId,
        u => u.Id,
        (t, u) => new { Task = t, UserName = u.Name })

var page = tasks
    .Skip((pageNumber - 1) * pageSize)
    .Take(pageSize)

var stats = new

### CRUD 操作

```
csharp
public class TaskRepository : ITaskRepository
{
    private readonly AppDbContext _context;

    public TaskRepository(AppDbContext context)
    {
        _context = context;
    }
    
    // Create
    public async Task<TaskItem> AddAsync(TaskItem task)
    {
        _context.Tasks.Add(task);
        await _context.SaveChangesAsync();
        return task;
    }
    
    // Read
    public async Task<List<TaskItem>> GetAllAsync()
    {
        return await _context.Tasks
            .Include(t => t.User)
            .OrderByDescending(t => t.CreatedAt)
            .ToListAsync();
    }
    
    public async Task<TaskItem?> GetByIdAsync(int id)
    {
        return await _context.Tasks.FindAsync(id);
    }
    
    // Update
    public async Task UpdateAsync(TaskItem task)
    {
        _context.Entry(task).State = EntityState.Modified;
        await _context.SaveChangesAsync();
    }
    
    // Delete
    public async Task DeleteAsync(int id)
    {
        var task = await _context.Tasks.FindAsync(id);
        if (task != null)
        {
            _context.Tasks.Remove(task);
            await _context.SaveChangesAsync();
        }
    }
    
    // 查询
    public async Task<List<TaskItem>> GetByCategoryAsync(string category)
    {
        return await _context.Tasks
            .Where(t => t.Category == category)
            .ToListAsync();
    }
}

public class TaskApiClient : ITaskApiClient

    public async Task<List<TaskItem>> GetTasksAsync()
    {

    {
        var json = JsonSerializer.Serialize(task);
        var content = new StringContent(json, Encoding.UTF8, "application/json");
        
        var response = await _httpClient.PutAsync($"/api/tasks/{id}", content);
        response.EnsureSuccessStatusCode();
    }
    
    // DELETE
    public async Task DeleteTaskAsync(int id)
    {
        var response = await _httpClient.DeleteAsync($"/api/tasks/{id}");
        response.EnsureSuccessStatusCode();
    }

## 11. 后台服务

### BackgroundService

```
csharp
public class TaskCleanupService : BackgroundService
{
    private readonly ILogger<TaskCleanupService> _logger;
    private readonly IServiceScopeFactory _scopeFactory;

    public TaskCleanupService(
        ILogger<TaskCleanupService> logger,
        IServiceScopeFactory scopeFactory)
    {
        _logger = logger;
        _scopeFactory = scopeFactory;
    }
    
    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        _logger.LogInformation("后台服务启动");
    
        while (!stoppingToken.IsCancellationRequested)
        {
            try
            {
                // 创建作用域以使用 Scoped 服务
                using var scope = _scopeFactory.CreateScope();
                var dbContext = scope.ServiceProvider.GetRequiredService<AppDbContext>();
    
                // 清理 30 天前的已完成任务
                var cutoffDate = DateTime.UtcNow.AddDays(-30);
                var oldTasks = await dbContext.Tasks
                    .Where(t => t.IsDone && t.CreatedAt < cutoffDate)
                    .ToListAsync(stoppingToken);
    
                dbContext.Tasks.RemoveRange(oldTasks);
                await dbContext.SaveChangesAsync(stoppingToken);
    
                _logger.LogInformation("清理了 {Count} 个旧任务", oldTasks.Count);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "后台服务执行出错");
            }
    
            // 每天执行一次
            await Task.Delay(TimeSpan.FromDays(1), stoppingToken);
        }
    
        _logger.LogInformation("后台服务停止");
    }
}

// 注册
builder.Services.AddHostedService<TaskCleanupService>();

public class RequestLoggingMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<RequestLoggingMiddleware> _logger;

    public RequestLoggingMiddleware(
        RequestDelegate next,
        ILogger<RequestLoggingMiddleware> logger)
    {
        _next = next;
        _logger = logger;
    }
    
    public async Task InvokeAsync(HttpContext context)
    {
        var startTime = DateTime.UtcNow;
        
        _logger.LogInformation("请求开始: {Method} {Path}", 
            context.Request.Method, 
            context.Request.Path);
    
        try
        {
            await _next(context);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "请求处理异常");
            throw;
        }
        finally
        {
            var duration = DateTime.UtcNow - startTime;
            _logger.LogInformation("请求完成: {Duration}ms", duration.TotalMilliseconds);
        }
    }
}

public class AsyncRelayCommand : ICommand
{
    private readonly Func<object?, Task> execute;
    private readonly Predicate<object?>? canExecute;
    private bool isExecuting;

    public event EventHandler? CanExecuteChanged;
    
    public AsyncRelayCommand(Func<object?, Task> execute, Predicate<object?>? canExecute = null)
    {
        this.execute = execute;
        this.canExecute = canExecute;
    }
    
    public bool CanExecute(object? parameter) => 
        !isExecuting && (canExecute?.Invoke(parameter) ?? true);
    
    public async void Execute(object? parameter)
    {
        if (isExecuting) return;
    
        isExecuting = true;
        CanExecuteChanged?.Invoke(this, EventArgs.Empty);
    
        try
        {
            await execute(parameter);
        }
        finally
        {
            isExecuting = false;
            CanExecuteChanged?.Invoke(this, EventArgs.Empty);
        }
    }
}
